    # BST

    二叉搜索树(BST)是树状数据结构中最基本的类型，通过BST我们可以将一个数据集以一定规律放置，这样我们就可以快速的按照逻辑访问我们需要的目标数据，大大降低了搜素数据的时间，我们可以通过一张图片大致了解一下二叉树的结构：



    我们通过BST的结构可知，每个整数都作为树的节点，以右侧节点值大于父节点，左侧节点值小于父节点的规律组合为一个可通过逻辑访问指定节点的一个整体。接下来就通过javascript代码一步步实现BST的各个功能： 
    function BinarySearchTree() {//二叉搜索树(BST)
    var Node = function(key){
        this.key = key;//节点的键
        this.left = null;//指向左侧键的指针
        this.right = null;//指向右侧键的指针
    };
}

    所谓属性就是类似于树的高度，树的种类，叶子是否茂盛之类的对于某类抽象事物的描述，而属性值就是这类事物的具体描述，例如树的高度为5米，树的种类为柏树，树的叶子很茂盛。而初始化对象就是将某类具体的事物指明为某个独立的个体，比如说：这棵树高度为5米，树的种类为柏树......而事物的类别class往往在javascript中以构造函数的形式被定义，这样就可以方便的把多个功能模块封装进对象了，我们初始化的对象tree就是事物类别的承载体。

    因为树的每个节点也可以作为对象，所以我们为此定义了构造函数Node这样就可以给每个节点赋予键key 左侧节点left 右侧节点right这三个属性。

    接下来我们就要编写连接这些节点的程序了：

    this.insert = function(key){//向树中插入一个新的键

        var newNode = new Node(key);

        if (root === null){//如果根节点为空
            root = newNode;//根节点为新的节点
        } else {
            insertNode(root,newNode);
        }
};

var insertNode = function(node, newNode){
    if (newNode.key < node.key){//如果新节点的键小根节点的键,则插入在树的左侧
        if (node.left === null){//如果左侧为空
            node.left = newNode;//直接插入
        } else {
            insertNode(node.left, newNode);//否则递归调用insertNode直至左侧节点的值为空
        }
    } else {//如果新节点的键大于根节点的键,则插入在树右侧
        if (node.right === null){
            node.right = newNode;
        } else {
            insertNode(node.right, newNode);
        }
    }
};

    这段代码都是写在构造函数BinarySearchTree内部的，其中insert属性(对象的属性也可以是某个动态的功能)的值为一个函数，在这个函数中我们初始化节点，然后根据节点的情况将节点插入到树的正确位置，这里的insertNode函数承担了这个任务。我们采用递归按照我们刚才通过图片了解到的规律安放节点。

    //在树中查找一个键，如果节点存在，则返回true；如果不存在，则返回false
this.search = function(key){
    return searchNode(root, key);
};
var searchNode = function(node, key){
    if (node === null){
         return false;
    }
    if (key < node.key){//目标值小于当前节点的值
        return searchNode(node.left, key);
    } else if (key > node.key){
        return searchNode(node.right, key);
    } else { //值等于当前节点的值,也就是要找的节点
        return true;
    }
};

var removeNode = function(node, element){

    if (node === null){//如果节点不存在则返回空
        return null;
    }

    if (element < node.key){//如果要删除的值比当前点的值小,则继续寻找左边的更小的值
        node.left = removeNode(node.left,element);//递归查询，直到找到相等值
        return node;

    } else if (element > node.key){//如果要删除的比当前节点的值大,则继续寻找右边的更大的值
        node.right = removeNode(node.right,element);
        return node;

    } else { //要删除的值与当前节点的值相比，不比当前点值大或小，则要删除的值就是当前节点的值

        //handle 3 special conditions
        //1 - a leaf node
        //2 - a node with only 1 child
        //3 - a node with 2 children

        //case 1
        if (node.left === null && node.right ===null){//外部节点，不包含任何子键
            node = null;//删除节点
            return node;
        }

        //case 2
        if (node.left === null){//只有右侧子节点在，要删除的节点成为右侧子节点的父节点，则父节被删除，子节点保留
            node = node.right;
            return node;

        } else if (node.right === null){
            node = node.left;
            return node;
        }
        //case 3
        //左边的节点和右侧的节点都存在，则当前节点的子点都要保留
        var aux = findMinNode(node.right);//得到前节点右侧节点(比当前节点的值大)的最小值，这个一定比当前节点的左侧第一个节点值大
        node.key = aux.key;//当前节点的值变为右侧树最小的值
        node.right = removeNode(node.right,aux.key);//由于最小节点已经求出，所以还需要查最小节点右侧是否存在一个只比最小节点大的子节(case 1,2),
        // 如果存在，则这个只比最小节点大的节点位置变之前最小节点的位置
        //此时原来右侧子树的最小节点成为这棵树的父节点原来只比最小节点大的，存在于最小节点右侧的节点置变为原来的最小节点位置，成为新的最小节点(如果在的话)
        return node;
    }
};

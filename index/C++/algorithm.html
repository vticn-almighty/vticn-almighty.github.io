# algorithm

头文件<algorithm>定义了一组专门设计用于元素范围的函数集合。

范围是可以通过迭代器或指针访问的任何对象序列，例如数组或某些STL容器的实例。但是请注意，算法通过迭代器直接对值进行操作，而不以任何方式影响任何可能容器的结构（它从不影响容器的大小或存储分配）。

int main()
{
  srand ( unsigned ( std::time(0) ) );
  array<int, 8> foo = {3, 5, 7, 17, 19, 5, 7, 23};
  array<int, 2> x = {5, 7};
  array<int, 10> y = {1, 8, 3, 3, 3, 2, 1, 1, 9, 6};
  array<int, 10> z = {1, 8, 3, 8, 3, 2, 1, 4, 9, 6};
  array<int, 8>::iterator begin = foo.begin(), end = foo.end();
  if (all_of(begin, end, [](int i) { return i % 2; }))
    cout << "所有元素均为奇数\n";
  if (any_of(begin, end, [](int i) { return i > 13; }))
    cout << "存在大于13的元素\n";
  if (none_of(begin, end, [](int i) { return i > 100; }))
    cout << "不存在大于100的元素\n";
  for_each(begin, end, [](int i) { cout << ' ' << i; });
  cout << endl;
  cout << *find(begin, end, 11) << endl;
  // 传入判断条件的函数筛选begin到end范围内的元素
  cout << *find_if(begin, end, [](int i) { return i > 17; }) << endl;
  // 从begin到end范围内的元素序列中寻找x子序列最后一次出现的位置，
  // 返回该位置的迭代器，未找到则返回序列末端元素的迭代器
  cout << *(find_end(begin, end, x.begin(), x.end()) - 1) << endl;
  // search与find_end类似，只不过前者返回第一次子序列x出现在目标序列中的位置
  cout << *(search(begin, end, x.begin(), x.end()) - 1) << endl;
  // 找到y序列中3连续出现两次的子序列首位置，返回迭代器
  cout << *(search_n(y.begin(), y.end(), 2, 3, comp)) << endl;
  // find_first_of从begin到end标记的目标序列的元素中匹配x.begin()到x.end()中
  // 的第一个相等元素，返回序列中符合条件的第一个元素的迭代器
  cout << *(find_first_of(begin, end, x.begin(), x.end(), comp) - 1) << endl;
  // 找到y序列中存在的第一组重复元素的位置，返回指向这个位置的迭代器
  array<int, 8>::iterator it = adjacent_find(y.begin(), y.end());
  cout << *(it - 1) << endl; //该位置之前的元素是3
  // 将迭代器指向重复元素的末端
  while (*it == *(it + 1))
    it++;
  cout << *(it + 1) << endl;                           //该位置之后的元素是2
  cout << count(y.begin(), y.end(), 3) << endl;        //计算序列中某元素出现的次数
  cout << count_if(y.begin(), y.end(), IsOdd) << endl; //计算序列中符合条件的元素个数
  pair<array<int, 10>::iterator, array<int, 10>::iterator> mypair;
  // 从目标序列y中查找z序列中不匹配的元素，以元素的迭代器对的形式返回
  mypair = mismatch(y.begin(), y.end(), z.begin());
  cout << "第一个不匹配元素对: " << *mypair.first << " - " << *mypair.second << endl;
  // 跳过第一对不匹配元素，从匹配剩下的序列中继续匹配
  mypair = mismatch(++mypair.first, y.end(), ++mypair.second);
  cout << "第二个不匹配元素对: " << *mypair.first << " - " << *mypair.second << endl;
  array<int, 4> n = {1, 8, 4, 8};
  array<int, 4> m = {8, 4};
  // 将两个队列中的元素进行对应比较，如果相等则返回true
  cout << (equal(n.begin() + 1, n.end() - 1, m.begin(), comp) ? "序列相等" : "序列不相等") << endl;
  n = {1, 3, 2, 4};
  m = {1, 2, 4, 3};
  cout << (is_permutation(n.begin(), n.end(), m.begin()) ? "n是m的一种排列" : "n不是m的一种排列") << endl;
  array<int, 10> r;
  //将m序列复制到r序列的指定位置，该序列上的数据会覆盖原来的数据，返回复制到目标序列的位置
  copy(m.begin(), m.end(), r.end() - 4);
  copy_if(m.begin(), m.end(), r.begin(), [](int i) { return i > 2; }); //根据条件进行复制
  for (array<int, 10>::iterator it = r.begin(); it != r.end(); it++)
    cout << *it << " ";
  cout << endl;
  // 将序列y中的元素以右对齐的形式复制到r序列中，覆盖r序列原有元素
  copy_backward(y.begin(), y.end(), r.end());
  for (array<int, 10>::iterator it = r.begin(); it != r.end(); it++)
    cout << *it << " ";
  cout << endl;
  array<int, 10> c;
  move(m.begin(), m.end(), c.end() - 5);
  for (array<int, 10>::iterator it = c.begin(); it != c.end(); it++)
    cout << *it << " ";
  cout << endl;
  for (array<int, 10>::iterator it = m.begin(); it != m.end(); it++)
    cout << *it << " ";
  cout << endl;
  vector<int> data{1, 2, 3, 4, 5, 6, 7, 8};
  // // 将data中的后6个元素移动到序列开始处，因为是移动所以原来的序列可能被扰乱，不应该继续使用原来的元素值
  // // move_backward类似，只是第三个参数是标记源序列右对齐的位置
  move(std::begin(data) + 2, std::end(data), std::begin(data));
  data.erase(std::end(data) - 2, std::end(data));
  copy(std::begin(data), std::end(data), std::ostream_iterator<int>{cout, " "});
  cout << endl;
  int t1 = 10, t2 = 20;
  swap(t1, t2); //也可以交换容器
  cout << "t1: " << t1 << " t2: " << t2 << endl;
  
  return 0;
}
